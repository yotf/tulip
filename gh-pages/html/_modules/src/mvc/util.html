    """Vraca ime izbora
    L1010->l i tako"""
    try:
        name = re.search(r'(L|THERM|MC|T)',string).group().lower()
        assert name in ['l','t','therm','mc']
    except:
        show_error('Wrong choice format','New format? Please contact developer!')
    else:
